CREATE TABLE essence (
	ID INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
	name VARCHAR(20) NOT NULL,
	CONSTRAINT pk_essence PRIMARY KEY (ID),
    CONSTRAINT uq_essence UNIQUE(name)
);

CREATE TABLE subtype (
	ID INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
	name VARCHAR(20) NOT NULL,
	CONSTRAINT pk_subtype PRIMARY KEY (ID),
    CONSTRAINT uq_subtype UNIQUE(name)
);

CREATE TABLE material (
	ID INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
	name VARCHAR(20) NOT NULL,
	essence1 INTEGER NOT NULL,
	amount1 DOUBLE PRECISION NOT NULL,
	essence2 INTEGER NOT NULL,
	amount2 DOUBLE PRECISION NOT NULL,
	subtype INTEGER NOT NULL,
	composite BOOLEAN NOT NULL,
	tier SMALLINT NOT NULL CHECK (tier IN (0, 1, 2, 3, 4)),
	CONSTRAINT pk_material PRIMARY KEY (ID),
	CONSTRAINT uq_material UNIQUE(name),
    CONSTRAINT fk_material_essence1 FOREIGN KEY (essence1) REFERENCES essence(ID),
    CONSTRAINT fk_material_essence2 FOREIGN KEY (essence2) REFERENCES essence(ID),
	CONSTRAINT fk_material_subtype FOREIGN KEY (subtype) REFERENCES subtype(ID)
);

CREATE TABLE recipe (
	ID INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
	name VARCHAR(20) NOT NULL,
	primary_amount1 DOUBLE PRECISION NOT NULL,
	primary_essence1 INTEGER NOT NULL,
	primary_amount2 DOUBLE PRECISION NOT NULL,
	primary_essence2 INTEGER NOT NULL,
	primary_amount3 DOUBLE PRECISION NOT NULL,
	primary_essence3 INTEGER NOT NULL,
	primary_amount4 DOUBLE PRECISION NOT NULL,
	primary_essence4 INTEGER NOT NULL,
	secondary_amount1 DOUBLE PRECISION NOT NULL,
	secondary_essence1 INTEGER NOT NULL,
	secondary_amount2 DOUBLE PRECISION NOT NULL,
	secondary_essence2 INTEGER NOT NULL,
	secondary_amount3 DOUBLE PRECISION NOT NULL,
	secondary_essence3 INTEGER NOT NULL,
	secondary_amount4 DOUBLE PRECISION NOT NULL,
	secondary_essence4 INTEGER NOT NULL,
	CONSTRAINT pk_recipe PRIMARY KEY (ID),
	CONSTRAINT uq_recipe UNIQUE(name),
        CONSTRAINT fk_recipe_pessence1 FOREIGN KEY (primary_essence1) REFERENCES essence(ID),
        CONSTRAINT fk_recipe_pessence2 FOREIGN KEY (primary_essence2) REFERENCES essence(ID),
        CONSTRAINT fk_recipe_pessence3 FOREIGN KEY (primary_essence3) REFERENCES essence(ID),
        CONSTRAINT fk_recipe_pessence4 FOREIGN KEY (primary_essence4) REFERENCES essence(ID),
        CONSTRAINT fk_recipe_sessence1 FOREIGN KEY (secondary_essence1) REFERENCES essence(ID),
        CONSTRAINT fk_recipe_sessence2 FOREIGN KEY (secondary_essence2) REFERENCES essence(ID),
        CONSTRAINT fk_recipe_sessence3 FOREIGN KEY (secondary_essence3) REFERENCES essence(ID),
        CONSTRAINT fk_recipe_sessence4 FOREIGN KEY (secondary_essence4) REFERENCES essence(ID)
);
